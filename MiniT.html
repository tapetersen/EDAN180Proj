BNF for MiniT.jj

TOKENS

/* SCANNER */

/* Skip whitespace */
 
<DEFAULT> SKIP : {
" "
| "\t"
| "\n"
| "\r"
| <":P" (~["\n"])* "\n">
}

   
/* Reserved words */
 
<DEFAULT> TOKEN : {
<PROCEDURE: "procedure">
| <BEGIN: "begin">
| <TYPE: "integer" | "bool">
| <FOR: "for">
| <UNTIL: "until">
| <DO: "do">
| <OD: "od">
| <IF: "if">
| <THEN: "then">
| <ELSE: "else">
| <FI: "fi">
| <END: "end">
| <RETURN: "return">
}

   
/* Literals */
 
<DEFAULT> TOKEN : {
<NUMERAL: (["0"-"9"])+>
| <BOOL: "true" | "false">
}

   
/* Operators */
 
<DEFAULT> TOKEN : {
<ASSIGN: "=">
| <MUL: "*">
| <ADD: "+">
| <SUB: "-">
| <DIV: "/">
| <GTE: ">=">
| <LTE: "<=">
| <LT: "<">
| <GT: ">">
| <EQ: "==">
| <NEQ: "!=">
| <AND: "and">
| <NOT: "not">
| <OR: "or">
}

   
/* separators */
 
<DEFAULT> TOKEN : {
<LPAR: "(">
| <RPAR: ")">
| <SEMI: ";">
| <COMMA: ",">
}

   
/* Identifiers */
 
<DEFAULT> TOKEN : {
<ID: (["A"-"Z","a"-"z"])+>
}

   
/* Illegal */
 
<DEFAULT> TOKEN : {
<ERROR: ~[]>
}

NON-TERMINALS

/* PARSER */
start ::= procedure <SEMI> <EOF>
procedure ::= proc_header proc_body
proc_header ::= <PROCEDURE> ( type )? id <LPAR> ( ( param ( <COMMA> declaration )* )? ) <RPAR>
proc_body ::= <BEGIN> stmts <END>
stmts ::= ( stmt <SEMI> )*
stmt ::= ( procedure | iddecl | assignment | expr | ifstmt | forstmt | retstmt )
ifstmt ::= <IF> expr <THEN> stmts ( <ELSE> stmts )? <FI>
forstmt ::= <FOR> stmt <UNTIL> expr <DO> stmts <OD>
retstmt ::= <RETURN> ( expr )?
param ::= declaration
iddecl ::= declaration
declaration ::= type id ( ( <ASSIGN> boolexpr )? )
assignment ::= iduse <ASSIGN> boolexpr
expr ::= boolexpr
boolexpr ::= boolterm ( <OR> boolterm )*
boolterm ::= relexpr ( <AND> relexpr )*
relexpr ::= numexpr ( <LT> numexpr | <GT> numexpr | <GTE> numexpr | <LTE> numexpr | <EQ> numexpr | <NEQ> numexpr )?
numexpr ::= term ( <ADD> term | <SUB> term )*
term ::= factor ( <MUL> factor | <DIV> factor )*
factor ::= funccall
| iduse
| numeral
| bool
| <LPAR> expr <RPAR>
| <SUB> factor
| <NOT> factor
funccall ::= id <LPAR> ( ( arglist )? ) <RPAR>
arglist ::= expr ( <COMMA> expr )*
numeral ::= <NUMERAL>
iduse ::= id
id ::= <ID>
type ::= <TYPE>
bool ::= <BOOL>
